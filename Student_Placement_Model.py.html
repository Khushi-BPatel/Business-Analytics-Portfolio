# t-tests for numeric features: compare means for Placement=1 vs Placement=0
placed = df[df["Placement"] == 1]
not_placed = df[df["Placement"] == 0]

rows = []
for col in numeric_cols:
    t_stat, p_val = stats.ttest_ind(
        placed[col].dropna(),
        not_placed[col].dropna(),
        equal_var=False
    )
    rows.append({"feature": col, "t_stat": t_stat, "p_value": p_val})

ttest_df = pd.DataFrame(rows).sort_values("p_value")
ttest_df

# Separate numeric & already-encoded binary columns
num_feats = [c for c in X_train.columns if c not in ["Internship_Experience"]]
bin_feats = [c for c in X_train.columns if c in ["Internship_Experience"]]

preprocess = ColumnTransformer(
    transformers=[
        ("num", StandardScaler(), num_feats),
        ("bin", "passthrough", bin_feats)
    ],
    remainder="drop"
)

log_reg = Pipeline(steps=[
    ("prep", preprocess),
    ("clf", LogisticRegression(max_iter=200, n_jobs=None))
])

log_reg.fit(X_train, y_train)
y_pred_lr = log_reg.predict(X_test)

print("Logistic Regression")
print("Accuracy:", accuracy_score(y_test, y_pred_lr))
print("Precision:", precision_score(y_test, y_pred_lr))
print("Recall:", recall_score(y_test, y_pred_lr))
print("F1:", f1_score(y_test, y_pred_lr))
print("\nConfusion Matrix:\n", confusion_matrix(y_test, y_pred_lr))
print("\nClassification Report:\n", classification_report(y_test, y_pred_lr))
